3.1 Procedimiento: Clasificación con KNN y Naive Bayes

Objetivo

Implementar un pipeline completo de clasificación con KNN y Naive Bayes, evaluando ambos modelos y comparando resultados.

Pasos

Paso 1: Preparar datos

library(caret); library(class); library(e1071)
set.seed(123)
data(iris)
idx <- createDataPartition(iris$Species, p=0.75, list=FALSE)
train <- iris[idx,]; test <- iris[-idx,]
cat("Train:", nrow(train), "| Test:", nrow(test), "
")

Paso 2: Normalizar features

preproc <- preProcess(train[,1:4], method=c("center","scale"))
train_n <- predict(preproc, train[,1:4])
test_n  <- predict(preproc, test[,1:4])
cat("Media post-normalización:
")
print(round(colMeans(train_n), 3))  # Debe ser ~0

Paso 3: Entrenar y evaluar KNN

pred_knn3 <- knn(train_n, test_n, train$Species, k=3)
pred_knn7 <- knn(train_n, test_n, train$Species, k=7)
cat("KNN k=3 accuracy:", mean(pred_knn3==test$Species), "
")
cat("KNN k=7 accuracy:", mean(pred_knn7==test$Species), "
")
print(table(pred_knn7, test$Species))  # Matriz de confusión

Paso 4: Entrenar y evaluar Naive Bayes

nb_model <- naiveBayes(Species ~ ., data=train)
pred_nb <- predict(nb_model, test)
cat("Naive Bayes accuracy:", mean(pred_nb==test$Species), "
")
print(table(pred_nb, test$Species))

Paso 5: Comparar modelos

resultados <- data.frame(
  Modelo = c("KNN k=3", "KNN k=7", "Naive Bayes"),
  Accuracy = c(
    mean(pred_knn3==test$Species),
    mean(pred_knn7==test$Species),
    mean(pred_nb==test$Species)
  )
)
print(resultados)

Paso 6: Validación cruzada robusta

ctrl <- trainControl(method="cv", number=10)
# KNN con CV
knn_cv <- train(Species~., data=iris, method="knn",
                trControl=ctrl,
                preProcess=c("center","scale"),
                tuneLength=10)
print(knn_cv)
plot(knn_cv)
# NB con CV
nb_cv <- train(Species~., data=iris, method="naive_bayes",
               trControl=ctrl)
cat("NB CV accuracy:", max(nb_cv$results$Accuracy), "
")
