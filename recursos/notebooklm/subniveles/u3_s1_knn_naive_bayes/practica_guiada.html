3.1 Práctica Guiada: Clasificación de Tumores (Benignos vs Malignos)

Dataset: Wisconsin Breast Cancer | Objetivo: clasificar tumores

Ficha 1: Cargar y explorar

library(mlbench); library(caret); library(class); library(e1071)
data(BreastCancer, package="mlbench")
# Limpiar NAs y convertir tipos
bc <- na.omit(BreastCancer)
bc[,2:10] <- lapply(bc[,2:10], function(x) as.numeric(as.character(x)))
cat("Dimensiones:", nrow(bc), "×", ncol(bc), "
")
table(bc$Class)  # ¿Cuántos benignos vs malignos?

Ficha 2: División y normalización

set.seed(42)
idx <- createDataPartition(bc$Class, p=0.8, list=FALSE)
train <- bc[idx,-1]; test <- bc[-idx,-1]  # Quitar columna ID
preproc <- preProcess(train[,1:9], method=c("center","scale"))
train_n <- cbind(predict(preproc, train[,1:9]), Class=train$Class)
test_n  <- cbind(predict(preproc, test[,1:9]),  Class=test$Class)
cat("Train:", nrow(train_n), "| Test:", nrow(test_n), "
")

Ficha 3: KNN con selección de K

k_vals <- c(1,3,5,7,9,11,15)
acc_knn <- sapply(k_vals, function(k) {
  p <- knn(train_n[,1:9], test_n[,1:9], train_n$Class, k=k)
  mean(p == test_n$Class)
})
data.frame(K=k_vals, Accuracy=round(acc_knn, 4))
k_mejor <- k_vals[which.max(acc_knn)]
cat("
Mejor K:", k_mejor, "| Accuracy:", max(acc_knn), "
")

Ficha 4: Naive Bayes y comparación

nb <- naiveBayes(Class ~ ., data=train_n)
pred_nb <- predict(nb, test_n)
pred_knn <- knn(train_n[,1:9], test_n[,1:9], train_n$Class, k=k_mejor)

cat("KNN accuracy:", mean(pred_knn==test_n$Class), "
")
cat("NB accuracy:", mean(pred_nb==test_n$Class), "
")

# Matriz de confusión de NB
cat("
Matriz NB:
")
print(table(Predicho=pred_nb, Real=test_n$Class))
