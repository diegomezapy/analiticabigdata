4.1 Práctica Guiada: Interpretando Importancia de Variables

Ficha 1: Entrenar Random Forest

library(randomForest); library(caret); library(vip)
data(wine, package="HDclassif")
wine$class <- factor(wine$class)
set.seed(77)
rf_wine <- randomForest(class~., data=wine, ntree=500, importance=TRUE)
print(rf_wine)

Ficha 2: Importancia por tipo

# Gini importance
importance(rf_wine, type=2)
# Accuracy permutation importance
importance(rf_wine, type=1)
# ¿Las dos métricas concuerdan en qué variables son más importantes?

Ficha 3: Visualizar

varImpPlot(rf_wine,
           main="Importancia de variables — Vinos",
           col="purple", pch=19)

# Con vip
vip(rf_wine, num_features=10,
    aesthetics=list(fill="#6A1B9A"))

Ficha 4: Modelo con top 5 variables

imp <- importance(rf_wine, type=1)
top5 <- rownames(imp)[order(imp[,1], decreasing=TRUE)][1:5]
cat("Top 5:", paste(top5, collapse=", "), "
")
formula_red <- as.formula(paste("class ~", paste(top5, collapse="+")))
rf_red <- randomForest(formula_red, data=wine, ntree=500)
cat("OOB error completo:", rf_wine$err.rate[500,"OOB"], "
")
cat("OOB error reducido:", rf_red$err.rate[500,"OOB"], "
")

¿El modelo con 5 variables mantiene un rendimiento comparable al de 13 variables?
