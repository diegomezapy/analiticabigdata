2.1 Procedimiento: EDA con ggplot2 paso a paso

Objetivo

Realizar un Análisis Exploratorio de Datos completo usando ggplot2 en R, siguiendo una secuencia sistemática para descubrir patrones en los datos.

Pasos

Paso 1: Preparar el entorno

library(ggplot2)
library(dplyr)
library(tidyr)
# Cargar datos de ejemplo
data(mtcars)
mtcars$cyl <- as.factor(mtcars$cyl)  # convertir a factor categórico

Paso 2: Vista general del dataset

str(mtcars)          # estructura y tipos
summary(mtcars)      # estadísticos descriptivos
colSums(is.na(mtcars))  # valores faltantes por columna

Paso 3: Análisis univariado

# Variable cuantitativa: histograma
ggplot(mtcars, aes(x = mpg)) +
  geom_histogram(bins = 15, fill = "#2196F3", color = "white") +
  labs(title = "Distribución: millas por galón", x = "mpg", y = "n") +
  theme_minimal()

# Variable categórica: barras
ggplot(mtcars, aes(x = cyl, fill = cyl)) +
  geom_bar() +
  scale_fill_brewer(palette = "Blues") +
  theme_minimal() + theme(legend.position = "none")

Paso 4: Análisis bivariado

# Cuantitativa vs cuantitativa
ggplot(mtcars, aes(x = wt, y = mpg)) +
  geom_point(size = 3, alpha = 0.7) +
  geom_smooth(method = "lm") +
  labs(title = "Peso vs Eficiencia", x = "Peso (1000 lb)", y = "mpg")

# Cuantitativa vs categórica
ggplot(mtcars, aes(x = cyl, y = mpg, fill = cyl)) +
  geom_boxplot() +
  labs(title = "Eficiencia por nro de cilindros")

Paso 5: Análisis multivariado

# Añadir tercera variable como color
ggplot(mtcars, aes(x = wt, y = mpg, color = cyl, size = hp)) +
  geom_point(alpha = 0.8) +
  scale_size_continuous(range = c(3, 10)) +
  labs(title = "Peso, eficiencia, cilindros y potencia")

# Facetas por variable categórica
ggplot(mtcars, aes(x = wt, y = mpg)) +
  geom_point() +
  facet_wrap(~ cyl, labeller = label_both) +
  theme_bw()

Paso 6: Guardar y documentar

grafico <- ggplot(mtcars, aes(x = wt, y = mpg, color = cyl)) +
  geom_point(size = 3) +
  labs(title = "EDA — mtcars",
       caption = "Fuente: Motor Trend 1974")
ggsave("eda_mtcars.png", grafico, width = 8, height = 6, dpi = 300)
