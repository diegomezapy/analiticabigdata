2.1 Aplicación: Visualización de Datos de Salud en Paraguay

Contexto

El Ministerio de Salud Pública y Bienestar Social de Paraguay publica datos epidemiológicos que pueden analizarse con las técnicas de visualización aprendidas. En este ejercicio, simulamos un dataset realista basado en indicadores de salud.

Dataset simulado

set.seed(42)
n <- 500
salud_py <- data.frame(
  departamento = sample(c("Central","Alto Paraná","Itapúa","Caaguazú","Amambay"), n, replace=TRUE),
  edad = round(rnorm(n, 35, 15)),
  peso_kg = round(rnorm(n, 70, 12)),
  talla_cm = round(rnorm(n, 165, 8)),
  glucosa = round(rnorm(n, 95, 20)),
  anio = sample(2019:2023, n, replace=TRUE)
)
salud_py$imc <- salud_py$peso_kg / (salud_py$talla_cm/100)^2
salud_py$categoria_imc <- cut(salud_py$imc,
  breaks = c(0, 18.5, 25, 30, Inf),
  labels = c("Bajo peso", "Normal", "Sobrepeso", "Obesidad"))

Tareas de visualización

Tarea 1: Distribución del IMC por departamento

ggplot(salud_py, aes(x = departamento, y = imc, fill = departamento)) +
  geom_violin(trim = FALSE, alpha = 0.7) +
  geom_boxplot(width = 0.1, fill = "white") +
  coord_flip() +
  theme_minimal() + theme(legend.position = "none") +
  labs(title = "Distribución del IMC por departamento",
       x = NULL, y = "IMC (kg/m²)")

Tarea 2: Relación edad-glucosa por categoría IMC

ggplot(salud_py, aes(x = edad, y = glucosa, color = categoria_imc)) +
  geom_point(alpha = 0.5, size = 2) +
  geom_smooth(method = "loess", se = FALSE) +
  facet_wrap(~ categoria_imc) +
  scale_color_brewer(palette = "Set1") +
  labs(title = "Edad vs Glucosa por categoría de IMC",
       caption = "Datos: simulación basada en MSPBS Paraguay")

Tarea 3: Evolución temporal

library(dplyr)
resumen_anual <- salud_py %>%
  group_by(anio, departamento) %>%
  summarise(imc_promedio = mean(imc), .groups = "drop")

ggplot(resumen_anual, aes(x = anio, y = imc_promedio,
                           color = departamento, group = departamento)) +
  geom_line(size = 1.2) +
  geom_point(size = 3) +
  scale_x_continuous(breaks = 2019:2023) +
  labs(title = "Evolución del IMC promedio por departamento (2019-2023)")
