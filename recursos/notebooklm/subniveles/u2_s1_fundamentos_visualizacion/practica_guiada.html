ggplot(gapminder, aes(x = lifeExp)) +
  geom_histogram(bins = 30, fill = "#0288D1") +
  facet_wrap(~ year, ncol = 4) +

Ficha 3: El famoso gráfico de Gapminder

gapminder_2007 <- gapminder %>% filter(year == 2007)
ggplot(gapminder_2007, aes(x = gdpPercap, y = lifeExp,
                            size = pop, color = continent)) +
  geom_point(alpha = 0.7) +
  scale_x_log10(labels = scales::comma) +
  scale_size_continuous(range = c(2, 15)) +
  labs(title = "Riqueza vs Salud (2007)",
       x = "PIB per cápita (escala log)", y = "Esperanza de vida")

Analiza: ¿Qué continente tiene más variabilidad en PIB? ¿Existe una relación clara?

Ficha 4: Evolución de América

americas <- gapminder %>%
  filter(continent == "Americas") %>%
  mutate(highlight = country == "Paraguay")

ggplot(americas, aes(x = year, y = lifeExp, group = country)) +
  geom_line(color = "gray70", alpha = 0.6) +
  geom_line(data = filter(americas, highlight), color = "red", size = 1.5) +
  labs(title = "Esperanza de vida — Américas (Paraguay en rojo)")

Reflexiona: ¿Cómo se ubica Paraguay respecto al continente?
