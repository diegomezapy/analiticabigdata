# ============ Diagnóstico inicial universal ============
diagnostico <- function(df) {
  cat("=== DIAGNÓSTICO DEL DATASET ===
")
  cat("Filas:", nrow(df), "| Columnas:", ncol(df), "
")
  cat("Tamaño en RAM:", format(object.size(df), units = "MB"), "
")
  cat("
Tipos de variable:
")
  print(sapply(df, class))
  cat("
Valores faltantes por columna:
")
  print(colSums(is.na(df)))
}
diagnostico(dt_csv)
