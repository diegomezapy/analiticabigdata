4.3 Práctica Guiada: Detectando y Mitigando Sesgo Algorítmico

Ficha 1: Crear dataset con sesgo deliberado

set.seed(2026)
n <- 600
contrataciones <- data.frame(
  experiencia_anos = round(rexp(n, 0.3)),
  habilidades_score = round(rnorm(n, 70, 15)),
  educacion = sample(c("Universitario","Postgrad","Tecnico"), n, replace=TRUE),
  sexo = factor(sample(c("M","F"), n, replace=TRUE))
)
# Sesgo: históricamente, hombres fueron contratados más
lp <- -2 + 0.3*contrataciones$experiencia_anos +
      0.04*contrataciones$habilidades_score +
      0.5*(contrataciones$educacion=="Postgrad") +
      1.2*(contrataciones$sexo=="M")  # SESGO: hombres favorecidos
contrataciones$contratado <- factor(rbinom(n,1,1/(1+exp(-lp))), labels=c("No","Si"))
table(contrataciones$sexo, contrataciones$contratado)

Ficha 2: Modelo con y sin variable sexo

library(caret)
set.seed(42)
idx <- createDataPartition(contrataciones$contratado, p=0.8, list=FALSE)
train <- contrataciones[idx,]; test <- contrataciones[-idx,]

# Modelo 1: incluye sexo
m1 <- glm(contratado ~ ., data=train, family=binomial)
# Modelo 2: excluye sexo
m2 <- glm(contratado ~ experiencia_anos + habilidades_score + educacion,
          data=train, family=binomial)
pred1 <- predict(m1, test, type="response")
pred2 <- predict(m2, test, type="response")

Ficha 3: Auditar ambos modelos

for (nombre in c("Con sexo", "Sin sexo")) {
  pred <- if(nombre=="Con sexo") pred1 else pred2
  pred_f <- factor(ifelse(pred>=0.5,"Si","No"), levels=c("No","Si"))
  cat(sprintf("
=== Modelo %s ===
", nombre))
  for (g in c("M","F")) {
    mask <- test$sexo == g
    tasa <- mean(pred_f[mask]=="Si")
    cat(sprintf("Grupo %s: tasa aprobacion = %.3f
", g, tasa))
  }
  tasa_g <- tapply(pred_f=="Si", test$sexo, mean)
  cat(sprintf("DIR = %.3f
", min(tasa_g)/max(tasa_g)))
}
