Paso 2: Calcular métricas por grupo

grupos <- unique(datos$sexo)
for (g in grupos) {
  subset_g <- test[test$sexo == g, ]
  pred_g <- predict(modelo, subset_g)
  real_g <- subset_g$resultado
  acc_g <- mean(pred_g == real_g)
  cat(sprintf("Grupo %s: n=%d, Accuracy=%.3f
", g, nrow(subset_g), acc_g))
}

Paso 3: Calcular Disparate Impact Ratio

tasa_positivos <- tapply(pred_test=="Si", test$sexo, mean)
dir <- min(tasa_positivos) / max(tasa_positivos)
cat(sprintf("DIR = %.3f (%s)
",
            dir,
            ifelse(dir < 0.8, "ALERTA: posible discriminacion", "OK")))

Paso 4: Calibrar umbrales por grupo

# Si el Recall difiere entre grupos, ajustar umbrales
# para equiparar la tasa de detección
probs <- predict(modelo, test, type="prob")[,"Si"]
for (g in unique(test$sexo)) {
  mask <- test$sexo == g
  umbral_g <- quantile(probs[mask & test$resultado=="No"], 0.7)
  cat(sprintf("Umbral sugerido para %s: %.3f
", g, umbral_g))
}
