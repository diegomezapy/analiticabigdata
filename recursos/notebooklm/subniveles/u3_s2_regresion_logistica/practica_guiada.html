3.2 Práctica Guiada: Predicción de Deserción Estudiantil

Contexto: Un sistema universitario quiere predecir qué estudiantes corren riesgo de desertar.

Ficha 1: Simular datos estudiantiles

set.seed(55)
n <- 500
estudiantes <- data.frame(
  promedio = round(rnorm(n, 65, 15), 1),
  ausencias = rpois(n, 3),
  distancia_km = round(rexp(n, 0.1)),
  trabaja = sample(c(0,1), n, replace=TRUE, prob=c(0.6,0.4)),
  beca = sample(c(0,1), n, replace=TRUE, prob=c(0.7,0.3)),
  primer_generacion = sample(c(0,1), n, replace=TRUE, prob=c(0.55,0.45))
)
# Variable objetivo: desercion (influenciada por promedio y ausencias)
lp <- -3 + 0.05*estudiantes$ausencias - 0.04*estudiantes$promedio +
      0.3*estudiantes$trabaja - 0.8*estudiantes$beca
prob_des <- 1/(1+exp(-lp))
estudiantes$desercion <- factor(rbinom(n,1,prob_des), labels=c("No","Si"))
table(estudiantes$desercion)

Ficha 2: Modelo de regresión logística

library(caret); library(pROC)
set.seed(42)
idx <- createDataPartition(estudiantes$desercion, p=0.8, list=FALSE)
train <- estudiantes[idx,]; test <- estudiantes[-idx,]
modelo <- glm(desercion ~ ., data=train, family=binomial)
summary(modelo)
# ¿Qué variables son significativas?
# ¿La beca disminuye el riesgo de deserción?

Ficha 3: Odds ratios e interpretación

or <- exp(coef(modelo))
ic <- exp(confint(modelo))
tabla_or <- cbind(OR=round(or,3), round(ic,3))
print(tabla_or)
# Interpreta: OR de 'trabaja' — ¿cuánto aumenta el odds de desertar si trabaja?
# Interpreta: OR de 'promedio' — ¿qué pasa por cada punto más de promedio?

Ficha 4: Evaluación y umbral

probs <- predict(modelo, test, type="response")
pred_05 <- factor(ifelse(probs>=0.5,"Si","No"), levels=c("No","Si"))
pred_03 <- factor(ifelse(probs>=0.3,"Si","No"), levels=c("No","Si"))
cm5 <- confusionMatrix(pred_05, test$desercion, positive="Si")
cm3 <- confusionMatrix(pred_03, test$desercion, positive="Si")
cat("Umbral 0.5 — Recall:", cm5$byClass["Sensitivity"], "
")
cat("Umbral 0.3 — Recall:", cm3$byClass["Sensitivity"], "
