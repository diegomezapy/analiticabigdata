2.2 Procedimiento: Limpieza de Datos Sistemática en R

Objetivo

Aplicar un pipeline de limpieza reproducible a un dataset con problemas típicos: NAs, duplicados, outliers y tipos incorrectos.

Pasos

Paso 1: Cargar y auditar

library(dplyr); library(ggplot2)
# Dataset con problemas intencionales
set.seed(1)
n <- 200
df <- data.frame(
  id = c(1:198, 10, 50),               # duplicados: id 10 y 50
  nombre = rep(LETTERS[1:5], 40),
  edad = c(rnorm(195, 30, 8), NA, NA, 200, -5),  # NAs y outliers
  ingreso = c(rexp(198, 1/3000000), NA, NA),
  categoria = c(sample(c("A","B","C"), 198, replace=TRUE), "x", "D")
)
str(df)
summary(df)

Paso 2: Valores faltantes

cat("NAs por columna:
")
print(colSums(is.na(df)))
# Visualizar patrón de NAs
library(naniar)
gg_miss_var(df)  # requiere naniar
# Imputar con mediana
df$edad[is.na(df$edad)] <- median(df$edad, na.rm=TRUE)
df$ingreso[is.na(df$ingreso)] <- median(df$ingreso, na.rm=TRUE)

Paso 3: Outliers

# Función genérica de detección
detectar_outliers <- function(x) {
  Q1 <- quantile(x, 0.25, na.rm=TRUE)
  Q3 <- quantile(x, 0.75, na.rm=TRUE)
  IQR_val <- Q3 - Q1
  x < Q1 - 1.5*IQR_val | x > Q3 + 1.5*IQR_val
}
cat("Outliers en edad:", sum(detectar_outliers(df$edad)), "
")
# Boxplot para visualizar
ggplot(df, aes(y = edad)) +
  geom_boxplot(fill = "#FF7043", outlier.color = "red") +
  ggtitle("Boxplot — edad (con outliers)")
# Reemplazar outliers con límites de Tukey (winsorizing)
cap_outliers <- function(x) {
  Q1 <- quantile(x, 0.25, na.rm=TRUE)
  Q3 <- quantile(x, 0.75, na.rm=TRUE)
  IQR_val <- Q3 - Q1
  pmax(pmin(x, Q3 + 1.5*IQR_val), Q1 - 1.5*IQR_val)
}
df$edad <- cap_outliers(df$edad)

Paso 4: Duplicados y tipos

cat("Filas duplicadas:", sum(duplicated(df$id)), "
")
df <- df[!duplicated(df$id), ]
df$nombre <- as.factor(df$nombre)
# Normalizar categoría
df$categoria <- toupper(df$categoria)  # "x" → "X"
df$categoria[!df$categoria %in% c("A","B","C")] <- NA

Paso 5: Reporte final

cat("Dataset limpio:
")
cat("  Filas:", nrow(df), "
")
cat("  NAs restantes:", sum(is.na(df)), "
")
summary(df)
