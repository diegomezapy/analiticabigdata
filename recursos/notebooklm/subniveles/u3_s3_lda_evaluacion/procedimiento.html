3.3 Procedimiento: Comparación Rigurosa de Modelos de Clasificación

Pasos

Paso 1: Preparar el benchmark

library(caret); library(MASS); library(e1071); library(class)
data(PimaIndiansDiabetes, package="mlbench")
set.seed(2024)
ctrl <- trainControl(method="cv", number=10,
                     savePredictions="final",
                     classProbs=TRUE)

Paso 2: Entrenar múltiples modelos

m_lda <- train(diabetes~., data=PimaIndiansDiabetes, method="lda",
               trControl=ctrl, preProcess=c("center","scale"))
m_knn <- train(diabetes~., data=PimaIndiansDiabetes, method="knn",
               trControl=ctrl, preProcess=c("center","scale"),
               tuneLength=10)
m_nb  <- train(diabetes~., data=PimaIndiansDiabetes, method="naive_bayes",
               trControl=ctrl)
m_lr  <- train(diabetes~., data=PimaIndiansDiabetes, method="glm",
               family="binomial", trControl=ctrl,
               preProcess=c("center","scale"))

Paso 3: Comparar

res <- resamples(list(LDA=m_lda, KNN=m_knn, NB=m_nb, LR=m_lr))
summary(res)
bwplot(res, metric="Accuracy")

Paso 4: Análisis del mejor modelo

# Suponer que LR fue el mejor
pred_final <- m_lr$pred
# Evaluación completa
library(pROC)
roc_obj <- roc(pred_final$obs, pred_final$pos)
cat("AUC:", round(auc(roc_obj), 3), "
")
plot(roc_obj, col="#2E7D32", lwd=2)
# Matriz de confusión
cm <- confusionMatrix(pred_final$pred, pred_final$obs, positive="pos")
print(cm)

Paso 5: Selección final y documentación

cat("Modelo seleccionado: Regresión Logística
")
cat("Justificación: mayor AUC-ROC en CV-10, interpretable, rápido
")
cat("AUC:", round(auc(roc_obj),3), "
")
cat("F1-Score:", round(cm$byClass["F1"],3), "
")
