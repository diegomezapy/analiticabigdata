# TABLA FORMATEADA (knitr + kableExtra)
library(knitr)
library(kableExtra)

tabla_resumen <- data.frame(
  Modelo = c("Regresión Logística", "KNN k=7", "Naive Bayes", "LDA"),
  AUC = c(0.847, 0.821, 0.789, 0.831),
  F1 = c(0.762, 0.741, 0.708, 0.750),
  Tiempo_ms = c(12, 580, 8, 15)
)

tabla_resumen %>%
  kbl(caption="Comparación de modelos de clasificación",
      digits=3,
      col.names=c("Modelo","AUC-ROC","F1-Score","Tiempo (ms)")) %>%
  kable_styling(bootstrap_options=c("striped","hover","condensed"),
                full_width=FALSE) %>%
  row_spec(1, bold=TRUE, background="#EDE7F6") %>%  # resaltar mejor
  footnote(general="Evaluación con 10-fold CV. Dataset: N=768.")

# GRÁFICO EJECUTIVO
library(ggplot2)
library(scales)

# Gráfico de barras con etiquetas de valor
ggplot(tabla_resumen, aes(x=reorder(Modelo, AUC), y=AUC, fill=Modelo)) +
  geom_col(alpha=0.85, width=0.7) +
  geom_text(aes(label=sprintf("%.3f", AUC)), hjust=-0.1, size=4.5) +
  scale_fill_brewer(palette="Purples") +
  coord_flip() +
  scale_y_continuous(limits=c(0, 1), labels=percent) +
  labs(
    title="AUC-ROC por modelo (10-fold CV)",
    subtitle="Mayor AUC = mejor capacidad de discriminación",
    x=NULL, y="AUC-ROC",
    caption="Fuente: Analítica de Datos ABC, 2026"
  ) +
  theme_minimal(base_size=13) +
  theme(legend.position="none",
        plot.title=element_text(face="bold"),
        plot.subtitle=element_text(color="gray50"))

# FORMATEO DE NÚMEROS PARA NEGOCIO
scales::dollar(1500000, prefix="Gs.", big.mark=".", decimal.mark=",")
# → "Gs.1.500.000"

# Renderizar el documento
rmarkdown::render("reporte_analisis.Rmd",
                  output_format="html_document",
                  output_file="reporte_analisis.html")

Python — Matplotlib + tabulate


import matplotlib.pyplot as plt
import matplotlib.patches as mpatches
import pandas as pd
import numpy as np

# GRÁFICO EJECUTIVO CON ANOTACIONES
modelos = ['Regresión
Logística', 'LDA', 'KNN k=7', 'Naive
Bayes']
aucs = [0.847, 0.831, 0.821, 0.789]
colores = ['#6A1B9A', '#9C27B0', '#CE93D8', '#EDE7F6']

fig, ax = plt.subplots(figsize=(9, 5))
bars = ax.barh(modelos, aucs, color=colores, height=0.6, edgecolor='white')

# Etiquetas de valor
for bar, auc in zip(bars, aucs):
    ax.text(auc + 0.005, bar.get_y() + bar.get_height()/2,
            f'{auc:.3f}', va='center', ha='left', fontsize=12, fontweight='bold')

# Línea de referencia aleatoria
ax.axvline(x=0.5, color='red', linestyle='--', alpha=0.5, label='Aleatorio (AUC=0.5)')

ax.set_xlim(0, 1.0)
ax.set_xlabel('AUC-ROC', fontsize=12)
ax.set_title('Comparación de Clasificadores
(Evaluación 10-fold CV)',
             fontsize=14, fontweight='bold')
ax.legend(loc='lower right')
ax.spines['top'].set_visible(False)
ax.spines['right'].set_visible(False)
ax.text(0.98, -0.12, 'Fuente: Analítica de Datos ABC, 2026',
        transform=ax.transAxes, ha='right', fontsize=9, color='gray')
plt.tight_layout()
plt.savefig('comparacion_modelos.png', dpi=150, bbox_inches='tight')

# TABLA FORMATEADA PARA REPORTE
from tabulate import tabulate
tabla = pd.DataFrame({
    'Modelo': modelos,
    'AUC': aucs,
    'Detecta 1 de cada N morosidades': [f'1 de {int(1/a*10):.0f}' for a in aucs]
})
print(tabulate(tabla, headers='keys', tablefmt='github', showindex=False))
