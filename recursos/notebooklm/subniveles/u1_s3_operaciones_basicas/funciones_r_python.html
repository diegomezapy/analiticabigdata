R — data.table + ggplot2


library(data.table)
library(ggplot2)

dt <- fread("historiales_salud.csv")

# ============ FILTRADO ============
# Filtrar pacientes adultos con diabetes
adultos_diabetes <- dt[edad >= 18 & diagnostico == "Diabetes"]

# ============ SELECCIÓN DE COLUMNAS ============
resumen_cols <- dt[, .(id_paciente, edad, diagnostico, region)]

# ============ ESTADÍSTICAS POR GRUPO ============
por_region <- dt[, .(
  n          = .N,
  edad_media = mean(edad, na.rm = TRUE),
  pct_diabet = mean(diagnostico == "Diabetes") * 100
), by = region]
print(por_region)

# ============ ORDENAR ============
por_region[order(-pct_diabet)]  # mayor a menor

# ============ VISUALIZACIÓN: Histograma ============
ggplot(dt, aes(x = edad)) +
  geom_histogram(binwidth = 5, fill = "#2196F3", color = "white") +
  labs(title = "Distribución de edad de pacientes",
       x = "Edad (años)", y = "Frecuencia") +
  theme_minimal()

# ============ VISUALIZACIÓN: Boxplot por diagnóstico ============
ggplot(dt, aes(x = diagnostico, y = tiempo_espera, fill = diagnostico)) +
  geom_boxplot(outlier.alpha = 0.3) +
  labs(title = "Tiempo de espera por diagnóstico",
       x = "Diagnóstico", y = "Tiempo de espera (minutos)") +
  theme_minimal() +
  theme(legend.position = "none")

# ============ VISUALIZACIÓN: Barras de frecuencia ============
ggplot(dt[, .(n = .N), by = region][order(-n)],
       aes(x = reorder(region, -n), y = n)) +
  geom_col(fill = "#4CAF50") +
  labs(title = "Pacientes por región", x = "Región", y = "N") +
  theme_minimal()

Python — Polars + Seaborn


import polars as pl
import seaborn as sns
import matplotlib.pyplot as plt

df = pl.read_csv("historiales_salud.csv")

# ============ FILTRADO ============
adultos_diabetes = df.filter(
    (pl.col("edad") >= 18) & (pl.col("diagnostico") == "Diabetes")
)

# ============ SELECCIÓN ============
resumen_cols = df.select(["id_paciente", "edad", "diagnostico", "region"])

# ============ ESTADÍSTICAS POR GRUPO ============
por_region = df.group_by("region").agg([
    pl.count().alias("n"),
    pl.col("edad").mean().alias("edad_media"),
    (pl.col("diagnostico") == "Diabetes").mean().alias("pct_diabet") * 100
]).sort("pct_diabet", descending=True)
print(por_region)

# ============ HISTOGRAMA ============
df_pd = df.to_pandas()  # seaborn usa pandas
plt.figure(figsize=(10, 5))
sns.histplot(data=df_pd, x="edad", bins=30, color="#2196F3")
plt.title("Distribución de edad de pacientes")
plt.xlabel("Edad (años)"); plt.ylabel("Frecuencia")
plt.tight_layout(); plt.show()

# ============ BOXPLOT ============
plt.figure(figsize=(10, 5))
sns.boxplot(data=df_pd, x="diagnostico", y="tiempo_espera", palette="Set2")
plt.title("Tiempo de espera por diagnóstico")
plt.tight_layout(); plt.show()
