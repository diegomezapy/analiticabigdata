Ficha 1: Estadísticas por grupo


library(data.table)
dt <- fread("historiales_salud.csv")

# Estadísticas por región
tabla_region <- dt[, .(
  pacientes = .N,
  edad_media = round(mean(edad), 1),
  pct_diabetes = round(mean(diagnostico == "Diabetes") * 100, 1),
  espera_mediana = median(tiempo_espera)
), by = region][order(-pacientes)]

print(tabla_region)

Ficha 2: Histograma de edades


library(ggplot2)
ggplot(dt, aes(x = edad)) +
  geom_histogram(binwidth = 5, fill = "#1976D2", color = "white", alpha = 0.8) +
  geom_vline(aes(xintercept = mean(edad)), color = "red", linetype = "dashed") +
  labs(title = "Distribución de edades",
       subtitle = paste("Media:", round(mean(dt$edad), 1), "años"),
       x = "Edad", y = "Frecuencia") +
  theme_minimal()

Pregunta: ¿La distribución es simétrica o sesgada? ¿Qué implica esto para el análisis de la media vs la mediana?
