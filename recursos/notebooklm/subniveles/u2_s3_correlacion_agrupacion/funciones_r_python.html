2.3 Guía de Funciones: Correlación y Agrupación

R — cor, dplyr, data.table


library(dplyr)
library(ggplot2)
library(data.table)

# CORRELACIÓN
# ─────────────
data(mtcars)

# Correlación entre dos variables
cor(mtcars$mpg, mtcars$wt)
# → retorna el coeficiente r de Pearson

# Correlación de Spearman (más robusta a outliers)
cor(mtcars$mpg, mtcars$wt, method = "spearman")

# Matriz de correlación completa
cor_matrix <- cor(mtcars[, c("mpg","wt","hp","disp")], use = "complete.obs")
# use="complete.obs": ignora NAs en pares

# Test de significancia
cor.test(mtcars$mpg, mtcars$wt)
# → r, t-statistic, p-value, IC 95%

# Visualizar matriz de correlación
library(corrplot)
corrplot(cor_matrix, method = "color", type = "upper",
         addCoef.col = "black", number.cex = 0.9,
         tl.cex = 0.9, col = colorRampPalette(c("blue","white","red"))(200))
# method: "circle","square","ellipse","number","shade","color"
# type: "full","upper","lower"
# addCoef.col: color de los números de correlación

# AGRUPACIÓN con dplyr
# ─────────────────────
mtcars %>%
  group_by(cyl) %>%
  summarise(
    n = n(),
    mpg_media = mean(mpg),
    mpg_sd = sd(mpg),
    hp_media = mean(hp),
    .groups = "drop"
  ) %>%
  arrange(desc(mpg_media))
# group_by: segmenta por nro de cilindros
# n(): cuenta filas en cada grupo
# .groups="drop": elimina agrupación después de summarise

# Agrupación múltiple
mtcars %>%
  group_by(cyl, am) %>%  # am: transmisión (0=automática, 1=manual)
  summarise(mpg_promedio = mean(mpg), .groups = "drop")

# AGRUPACIÓN con data.table (más rápido para big data)
dt <- as.data.table(mtcars)
resultado <- dt[, .(
  n = .N,
  mpg_media = mean(mpg),
  hp_media = mean(hp)
), by = .(cyl, am)]
# .N: número de filas del grupo
# by: variables de agrupación

# TABLA DE CONTINGENCIA
table(mtcars$cyl, mtcars$am)
# Test chi-cuadrado de independencia
chisq.test(table(mtcars$cyl, mtcars$am))
# p < 0.05: hay asociación entre nro cilindros y tipo de transmisión

Python — pandas + scipy


import pandas as pd
import numpy as np
from scipy import stats
import seaborn as sns
import matplotlib.pyplot as plt

df = pd.read_csv('https://raw.githubusercontent.com/mwaskom/seaborn-data/master/tips.csv')

# CORRELACIÓN
df[['total_bill','tip','size']].corr()            # Pearson
df[['total_bill','tip','size']].corr(method='spearman')  # Spearman

# Test de significancia
r, p = stats.pearsonr(df['total_bill'], df['tip'])
print(f"r = {r:.3f}, p = {p:.4f}")

# Heatmap
fig, ax = plt.subplots(figsize=(6,5))
sns.heatmap(df[['total_bill','tip','size']].corr(),
            annot=True, fmt='.2f', cmap='coolwarm',
            center=0, vmin=-1, vmax=1, ax=ax)
plt.title('Matriz de correlación')
plt.tight_layout()

# AGRUPACIÓN
# Por un grupo
df.groupby('day')['total_bill'].agg(['mean','std','count'])

# Por múltiples grupos
resumen = df.groupby(['day','sex']).agg(
    promedio_cuenta=('total_bill', 'mean'),
    propina_media=('tip', 'mean'),
    n=('total_bill', 'count')
).reset_index()
# .agg(): agrega con nombre personalizado
# reset_index(): convierte índices jerárquicos en columnas

# TABLA DE CONTINGENCIA
tabla = pd.crosstab(df['day'], df['sex'])
# Test chi-cuadrado
chi2, p, dof, expected = stats.chi2_contingency(tabla)
print(f"chi2={chi2:.2f}, p={p:.4f}, gl={dof}")
